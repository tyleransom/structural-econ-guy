\textcolor{navy}{Central Limit Theorem}

\bigskip{}

Sample mean converges to normal: $\overline{X}_N \xrightarrow{d} N\left(\mu, \frac{\sigma^2}{N}\right)$

\bigskip{}

\begin{itemize}
\itemsep1.5em
\item<2-> Simulation: $N = 100$, $B = 1{,}000{,}000$ replications, $X\sim$ Exponential$(2.0)$
\item<3-> Compute sample mean for each replication
\item<4-> Overlay histogram of sampling distribution with fitted normal density
\end{itemize}

\end{frame}

\textcolor{navy}{Extreme Value Theory}

\bigskip{}

\begin{itemize}
\itemsep1.5em
\item<2-> Show $\displaystyle M_N = \max_i \left\{X_i\right\}_{i=1}^N$ converges to Gumbel
\item<3-> If $X\sim$ Pareto instead of Exponential, show convergence to Fr\'{e}chet
\item<4-> Show $\displaystyle m_N = \min_i \left\{X_i\right\}_{i=1}^N$ converges to Exponential if $X\sim$ Exponential
\end{itemize}
